        <CardContent className="flex flex-wrap items-end gap-3">

              id="symbol-input"
              data-testid="aggregate-symbol-input"
              value={symbol}
              onChange={(e) => setSymbol(e.target.value)}
              onKeyDown={(e) => { if (e.key === "Enter") handleGo(); }}
              placeholder="BTCUSDT"
              className="h-10 w-[200px]"

            data-testid="aggregate-symbol-go"

      </Card>

```tsx
export const Route = createFileRoute("/shadow/aggregate/$symbol")({
  component: ShadowAggregateSymbolPage,
});

const WINDOW_30D_MS = 30 * 24 * 60 * 60 * 1000;

interface Filters {
  botId: string;
  range: TimeRange;
}

function buildAggregateUrl(symbol: string, filters: Filters): string {
  const params: string[] = [];
  if (filters.botId !== "") {
    params.push(`bot_id=${encodeURIComponent(filters.botId)}`);
  }
  // Time range always applies (created_at non-null per migration 0014; mirror
  // T-517b2 shadow.rejected.tsx WG#4 inversion of paper-trades omit-when-active).
  params.push(`from=${encodeURIComponent(filters.range.from.toISOString())}`);
  params.push(`to=${encodeURIComponent(filters.range.to.toISOString())}`);
  return `/api/shadow/aggregate/${encodeURIComponent(symbol)}?${params.join("&")}`;
}

function ShadowAggregateSymbolPage(): React.JSX.Element {
  const { symbol } = Route.useParams();
  const [filters, setFilters] = React.useState<Filters>(() => {
    const now = new Date();
    return {
      botId: "",
      range: { from: new Date(now.getTime() - WINDOW_30D_MS), to: now, preset: "30d" },
    };
  });

  const aggregateQuery = useQuery({
    queryKey: ["shadow-aggregate", symbol, filters],
    queryFn: () =>
      apiFetch<VariantAggregateListResponse>(buildAggregateUrl(symbol, filters)),
  });

  const variants = aggregateQuery.data?.variants ?? [];
  // ... filter card + table render ...
}

const VARIANT_COLUMNS: ColumnDef<VariantAggregateResponse, unknown>[] = [
  {
    accessorKey: "variant_name",
    header: "Variant",
    cell: ({ row }) => {
      // Best pill on first row only (sorted DESC by total_pnl per backend).
      const isBest = row.index === 0;
      return (
        <span className="inline-flex items-center gap-2">
          <span className="font-mono text-xs">{row.original.variant_name}</span>
          {isBest && <BestPill />}
        </span>
      );
    },
  },
  { accessorKey: "n_trades", header: "Trades" },
  { accessorKey: "win_count", header: "Wins" },
  {
    accessorKey: "win_rate",
    header: "Win %",
    cell: ({ row }) => formatPctNumber(row.original.win_rate),
  },
  {
    accessorKey: "total_pnl",
    header: "Total P&L",
    cell: ({ row }) => <PriceDelta value={row.original.total_pnl} />,
  },
  // ... avg_pnl + best_pnl + worst_pnl as PriceDelta ...
  // ... avg_mfe_pct + avg_mae_pct as formatPct (mirror ShadowVariantsView convention) ...
];

function BestPill(): React.JSX.Element {
  return (
    <span
      data-testid="aggregate-best-pill"
      className={
        "inline-flex items-center rounded-md px-1.5 py-0.5 text-[10px] font-medium " +
        "bg-primary/15 text-primary"
      }
    >
      Best
    </span>
  );
}

function formatPct(n: number | null): React.JSX.Element {
  if (n === null) {
    return <span className="text-muted-foreground">—</span>;
  }
  const colour = n >= 0 ? "text-green-400" : "text-red-400";
  return <span className={`font-mono text-xs ${colour}`}>{`${(n * 100).toFixed(2)}%`}</span>;
}

function formatPctNumber(n: number): React.JSX.Element {
  // Plain percent format for win_rate (no sign-coloring; always positive ratio).
  return <span className="font-mono text-xs">{`${(n * 100).toFixed(1)}%`}</span>;
}
```

## Edge cases

1. **Index page: empty input** — Go button disabled when `symbol.trim() === ""`. Pressing Enter on empty input is a no-op.
2. **Index page: symbol with whitespace** — `symbol.trim()` before navigate; preserves operator-typed casing.
3. **Symbol param URL-encoded** — `encodeURIComponent(symbol)` in `buildAggregateUrl` (defensive; symbols are typically alphanumeric but defensive against future weird symbols).
4. **Detail page: empty variants result** — `aggregateQuery.data?.variants = []` → DataTable renders `"No variants for $symbol in selected window"` empty message; symbol echoed in message for orientation.
5. **Detail page: API 404 / 5xx** — `aggregateQuery.isError` → red-text fallback.
6. **Detail page: backend filters via created_at, range always applies** — mirror T-517b2 WG#4 inversion (created_at non-null per migration 0014; NO omit-when-X heuristic).
7. **Best pill on single-variant case** — first (only) row gets pill; harmless visually.
8. **Best pill on empty case** — no rows → no pill; empty message renders.
9. **Decimal display** — `total_pnl/avg_pnl/best_pnl/worst_pnl` come as JSON strings; `<PriceDelta value={string}>` accepts string input (existing component contract).
10. **MFE/MAE % rendering** — plain text `${(n * 100).toFixed(2)}%` mirror `ShadowVariantsView.formatPctPair` + T-517b2 `formatPct` convention. NOT `<PriceDelta>` (statistical ratios, not money). Sign-color via Tailwind class.
11. **Win rate display** — `formatPctNumber` (plain percent without sign-color; always non-negative ratio in [0, 1]).
