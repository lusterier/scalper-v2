### Sub-commit #1 — Live-mode safeguard (`feat(T-520-livemode)`)

Per BRIEF §16.5:2253-2257 verbatim:

> Starting a bot with `exchange.mode: live` requires env var `BOT_CONFIRM_LIVE=yes` in the service's environment.
> Startup logs a loud warning `LIVE MODE ENGAGED` and sends a Telegram alert.
> Testnet and paper modes do not require this.

**Plug-point:** `services/execution/app/pool.py:_construct_bybit_adapter` is called from `register_adapters` for every `bot_row.exchange_mode in ("live", "testnet")`. The safeguard check happens BEFORE the `_construct_bybit_adapter` call when mode=="live"; raises `RuntimeError` to kill lifespan startup if env var not set.

**Implementation:**

```python
# services/execution/app/pool.py — NEW _check_live_mode_safeguard helper
_LIVE_MODE_CONFIRM_ENV = "BOT_CONFIRM_LIVE"
_LIVE_MODE_CONFIRM_VALUE = "yes"


async def _check_live_mode_safeguard(
    *,
    bot_row: BotRow,  # from packages.db.queries.execution per existing pool.py:53,151,190 imports
    env: Mapping[str, str],
    bus: NatsClient,
    bound_logger: BoundLogger,
    now_fn: Callable[[], datetime] = lambda: datetime.now(UTC),
) -> None:
    """BRIEF §16.5 safeguard — fail-fast if BOT_CONFIRM_LIVE!=yes for live mode.

    Logs `LIVE MODE ENGAGED` warning AND publishes NATS `system.alerts`
    message (consumed by alerting-svc T-409 catch-all rule → Telegram).
    Testnet + paper modes bypass this check (early return).

    Raises RuntimeError if exchange_mode=='live' AND env var unset/!=yes
    — fail-fast death of lifespan; operator sees clear error in logs.
    """
    if bot_row.exchange_mode != "live":
        return
    confirm = env.get(_LIVE_MODE_CONFIRM_ENV, "").strip().lower()
    if confirm != _LIVE_MODE_CONFIRM_VALUE:
        msg = (
            f"bot {bot_row.bot_id!r} configured for LIVE mode but "
            f"{_LIVE_MODE_CONFIRM_ENV}!={_LIVE_MODE_CONFIRM_VALUE} "
            f"(BRIEF §16.5 safeguard); refusing to construct live adapter"
        )
        bound_logger.error("live_mode_safeguard_failed", bot_id=bot_row.bot_id, env_value=env.get(_LIVE_MODE_CONFIRM_ENV))
        raise RuntimeError(msg)
    bound_logger.warning("LIVE MODE ENGAGED", bot_id=bot_row.bot_id, exchange_mode="live")
    # BRIEF §16.5:2256 verbatim — "AND sends a Telegram alert".
    # Publishes to `system.alerts` (alerting-svc catch-all routes per
    # configs/alerts.yaml:32-35 → channel=system → default template → Telegram).
    envelope = MessageEnvelope(
        schema_version="1.0",
        published_at=now_fn(),
        correlation_id=f"live-mode-{bot_row.bot_id}",
        payload={
            "event": "live_mode_engaged",
            "bot_id": bot_row.bot_id,
            "exchange_mode": "live",
        },
    )
    await bus.publish("system.alerts", envelope)
```

Then in `register_adapters` loop:

```python
if bot_row.exchange_mode in ("live", "testnet"):
    await _check_live_mode_safeguard(  # NEW per BRIEF §16.5
        bot_row=bot_row,
        env=resolved_env,
        bus=bus,
        bound_logger=bound_logger,
    )
    adapter, ws_task = _construct_bybit_adapter(...)
```

# T-520 sub-commit #2 — populate feature_history for both T-303 series
# conditions (RisingCondition + FallingCondition use `n_samples`) AND
# T-305 plugin Rules wrapped in PluginCondition (oi_squeeze uses
# `lookback_candles` on the inner Rule). Explicit fallback chain rather
# than getattr-by-attr-name sniff to make plugin path traceable.
def _required_history_window(condition: object) -> int | None:
    """Detect history window size from condition shape (T-303 + T-305 plugin)."""
    # T-303 RisingCondition + FallingCondition.
    n_samples = getattr(condition, "n_samples", None)
    if isinstance(n_samples, int):
        return n_samples
    # T-305 PluginCondition wrapping a Rule with `lookback_candles` (oi_squeeze).
    rule = getattr(condition, "rule", None)
    if rule is not None:
        lookback = getattr(rule, "lookback_candles", None)
        if isinstance(lookback, int):
            return lookback
    return None


# Inside _evaluate_rule, AFTER resolver_result + BEFORE RuleContext:
n_history = _required_history_window(rule.condition)
feature_history: dict[str, Sequence[FeatureValue]] = {}
if n_history is not None:
    history = await resolver.resolve_history(
        rule_feature=rule.feature,
        signal=signal,
        n_samples=n_history,
    )
    if history:
        feature_history[resolved_ref] = history

ctx = RuleContext(
    signal=signal,
    feature_snapshot={resolved_ref: resolver_result.value},
    feature_ref=resolved_ref,
    feature_history=feature_history,  # NEW
)
