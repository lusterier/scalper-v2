## Public interface

```tsx
// ui/src/components/YamlDiffView.tsx

import * as React from "react";
import { cn } from "@/lib/utils";

export interface YamlDiffViewProps {
  current: string;
  draft: string;
  className?: string;
}

type DiffOp =
  | { kind: "equal"; aLine: string; bLine: string }
  | { kind: "added"; bLine: string }
  | { kind: "removed"; aLine: string }
  | { kind: "modified"; aLine: string; bLine: string };

function diffLines(a: string[], b: string[]): DiffOp[] {
  // Pass 1: O(a×b) LCS DP returns ordered ops as raw {equal, added, removed}.
  // For typical bot config <200 lines, ~40k cell DP table fits in microseconds.
  //
  // Pass 2 (modify-collapse): scan ops; if op[i] is "removed" AND op[i+1] is
  // "added" (or vice versa) AND both are isolated (op[i-1] is "equal" or
  // out-of-bounds AND op[i+2] is "equal" or out-of-bounds — i.e., NOT part of
  // a longer contiguous remove/add block), collapse the pair into a single
  // "modified" op carrying both aLine + bLine. This produces cleaner UX:
  // a single-line edit shows as one yellow row instead of separate red+green
  // rows. Multi-line replacements (e.g., 3 removed + 3 added contiguous) stay
  // as separate ops — represents true insertion/deletion at scale.
  // ... implementation ...
}

export function YamlDiffView({
  current,
  draft,
  className,
}: YamlDiffViewProps): React.JSX.Element {
  const ops = React.useMemo(() => {
    if (current.trim() === "") {
      return draft.split("\n").map((line) => ({ kind: "added", bLine: line }) as DiffOp);
    }
    return diffLines(current.split("\n"), draft.split("\n"));
  }, [current, draft]);
  // Render side-by-side: left = "Current" column with removed+modified+equal lines;
  // right = "New (editing)" column with added+modified+equal lines.
  return (
    <div className={cn("grid grid-cols-2 gap-2 text-xs", className)} data-testid="yaml-diff-view">
      {/* ... */}
    </div>
  );
}
```

CSS classes (Tailwind utility):
- `bg-green-50` (added line in draft column).
- `bg-red-50` (removed line in current column).
- `bg-yellow-50` (modified line on both columns).
- `bg-transparent` (equal lines).

## Hand verification

### §A — Identical input

`current = draft = "a\nb\nc"`. Output: 3 equal ops, no highlight.

### §B — Pure addition

`current = "a\nb"`, `draft = "a\nb\nc"`. Output: 2 equal + 1 added (`c`).

### §C — Pure deletion

`current = "a\nb\nc"`, `draft = "a\nb"`. Output: 2 equal + 1 removed (`c`).

### §D — Modification (single line)

`current = "a\nb\nc"`, `draft = "a\nB\nc"`. Output: 1 equal + 1 modified (a:`b` → b:`B`) + 1 equal.

### §E — First-version (empty current)

`current = ""`, `draft = "a\nb"`. Output: 2 added ops (entire draft as added).

### §F — Empty draft

`current = "a\nb"`, `draft = ""`. Output: 2 removed ops.

### §G — Labels render

Component renders "Current" + "New (editing)" column headers (preserve T-416 labels).

### §H — Mixed 5-line scenario

`current = "alpha\nbeta\ngamma\ndelta\nepsilon"`, `draft = "alpha\nBETA\ndelta\nepsilon\nzeta"`.
Expected: equal `alpha` + modified `beta`→`BETA` + removed `gamma` + equal `delta` + equal `epsilon` + added `zeta`.
