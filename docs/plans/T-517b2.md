```tsx
export const Route = createFileRoute("/shadow/rejected")({
  component: ShadowRejectedPage,
});

const PAGE_SIZE = 50;
const WINDOW_30D_MS = 30 * 24 * 60 * 60 * 1000;

type StatusFilter = "all" | "active" | "terminated";
type TerminalOutcomeFilter = "all" | ShadowRejectedTerminal;

interface Filters {
  botId: string;
  symbol: string;
  status: StatusFilter;
  terminalOutcome: TerminalOutcomeFilter;
  range: TimeRange;
}

function buildShadowRejectedUrl(filters: Filters, limit: number, offset: number): string {
  const params: string[] = [];
  if (filters.botId !== "") params.push(`bot_id=${encodeURIComponent(filters.botId)}`);
  if (filters.symbol !== "") params.push(`symbol=${encodeURIComponent(filters.symbol)}`);
  if (filters.status !== "all") params.push(`status=${filters.status}`);
  if (filters.terminalOutcome !== "all") {
    params.push(`terminal_outcome=${filters.terminalOutcome}`);
  }
  // Backend filters created_at column (always non-null per migration 0014);
  // unlike paper-trades closed_at, NO omit-when-active heuristic needed.
  params.push(`from=${encodeURIComponent(filters.range.from.toISOString())}`);
  params.push(`to=${encodeURIComponent(filters.range.to.toISOString())}`);
  params.push(`limit=${String(limit)}`);
  params.push(`offset=${String(offset)}`);
  return `/api/shadow/rejected/?${params.join("&")}`;
}

function ShadowRejectedPage(): React.JSX.Element {
  const [filters, setFilters] = React.useState<Filters>(() => {
    const now = new Date();
    return {
      botId: "",
      symbol: "",
      status: "all",
      terminalOutcome: "all",
      range: { from: new Date(now.getTime() - WINDOW_30D_MS), to: now, preset: "30d" },
    };
  });
  const [page, setPage] = React.useState(0);
  const offset = page * PAGE_SIZE;

  const query = useQuery({
    queryKey: ["shadow-rejected-list", filters, offset],
    queryFn: () =>
      apiFetch<ShadowRejectedListResponse>(buildShadowRejectedUrl(filters, PAGE_SIZE, offset)),
  });

  // Filter card (BotSelector + symbol Input + status select + terminal_outcome
  // select + TimeRangePicker) → DataTable with REJECTED_COLUMNS → pagination
  // block. NO row click navigate per OQ-1 list-only.
}

const REJECTED_COLUMNS: ColumnDef<ShadowRejected, unknown>[] = [
  { accessorKey: "id", header: "#" },
  { accessorKey: "bot_id", header: "Bot" },
  { accessorKey: "symbol", header: "Symbol" },
  { accessorKey: "would_side", header: "Side" },
  {
    accessorKey: "terminated_at",
    header: "Status",
    cell: ({ row }) => (
      <span className={...}>
        {row.original.terminated_at === null ? "active" : "terminated"}
      </span>
    ),
  },
  {
    accessorKey: "terminal_outcome",
    header: "Outcome",
    cell: ({ row }) => row.original.terminal_outcome ?? "—",
  },
  {
    accessorKey: "mfe_pct",
    header: "MFE %",
    cell: ({ row }) =>
      row.original.mfe_pct !== null
        ? <PriceDelta value={(row.original.mfe_pct * 100).toFixed(2)} />
        : "—",
  },
  {
    accessorKey: "mae_pct",
    header: "MAE %",
    cell: ({ row }) =>
      row.original.mae_pct !== null
```

## Edge cases

1. **Empty filter result** — `query.data?.rejected = []`; DataTable renders `"No rejected signals match filters"` empty message.
2. **Backend 422 on garbage status / terminal_outcome** — UI never sends garbage (Filters typed `"all" | "active" | "terminated"` and `"all" | ShadowRejectedTerminal`); `query.isError` triggers red-text fallback.
3. **Pagination beyond total** — Next button disabled when `(page + 1) * PAGE_SIZE >= total`; page state preserved across filter changes (or reset on filter change to avoid empty pages — mirror paper-trades-index `setPage(0)` after filter mutation).
4. **`status=active` + terminal_outcome filter** — composable; backend AND-combines. UI sends both query params. Active rows have `terminal_outcome=null` so result will be empty unless the filter combination is logically possible (e.g. `terminated + would_tp` → terminated rows with that outcome). Operator-visible behavior; no special UI guard.
5. **Active row terminal_outcome rendering** — `terminal_outcome=null` for active rows → render `"—"` per existing convention (mirror paper-trades exit_price null case).
6. **Active row MFE/MAE rendering** — `mfe_pct=null` / `mae_pct=null` for active observations (T-513 only writes terminal MFE/MAE on observation close) → render `"—"`.
7. **Time range default** — last 30 days (mirror paper-trades / trades default); `WINDOW_30D_MS = 30 * 24 * 60 * 60 * 1000`.
8. **Time range always applies** — unlike paper-trades (omit `?from`/`?to` when status=open since closed_at NULL filters out range), `created_at` is always non-null per migration 0014 → range always applies; NO omit heuristic. Documented in `buildShadowRejectedUrl` comment.
9. **TanStack Router routeTree.gen.ts regen** — dev-server / build step regenerates; no manual edit. Test imports `routeTree` to confirm registration.
10. **PriceDelta for percent values** — `mfe_pct`/`mae_pct` are float fractions (e.g. 0.012 = 1.2%); render via `PriceDelta` with `(value * 100).toFixed(2)` per existing percent-display convention OR plain percent string. Plan defers exact formatting to impl-time mirror existing percent rendering in shadow_variants component.
