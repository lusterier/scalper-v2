### T-110c `pipeline.py` patch (Q11) — split `start()`

```python
def acquire_handles(self) -> None:
    """Synchronous: acquire BufferHandle per registered key.

    Split out of the prior monolithic start() so a composition root
    (T-110d) can sequence: acquire → warmup_load → start_consuming.
    Without this split, warmup pushes silent-drop on un-allocated
    buffers OR race live frames and wedge out-of-order data.
    """
    if not self._features_by_key:
        return
    for key in self._features_by_key:
        symbol, interval = key
        self._handles[key] = self._buffer_registry.acquire(symbol, interval)


async def start_consuming(self) -> None:
    """Async: subscribe to market.ohlc.1m.>.

    Caller must invoke acquire_handles() first; warmup_load (if any)
    must run between the two.
    """
    if not self._features_by_key:
        self._logger.info("feature_pipeline_started_empty")
        return
    await self._bus.subscribe(_SUBJECT_WILDCARD, self._on_envelope)
    self._logger.info("feature_pipeline_started", keys=sorted(self._features_by_key))
```

### `services/feature_engine/app/warmup.py`

```python
async def warmup_load(
    *,
    pool: asyncpg.Pool[asyncpg.Record],
    registry: BufferRegistry,
    features_by_key: Mapping[tuple[str, str], list[tuple[str, Feature]]],
    source: str,
    logger: BoundLogger,
) -> None:
    """Prime registry with cagg history for every registered key.

    Per-key error isolation: a fetch_warmup_window failure on key A
    logs feature_warmup_load_error and continues to key B —
    under-fill on A is recoverable on subsequent live frames
    (Feature.compute raises FeatureUnderflowError until enough
    candles accumulate).

    OhlcRow tuple-field order assumption (T-110b Write-time guidance #3):
    (symbol, bucket_start, open, high, low, close, volume, source).
    Verified by test_warmup_load_ohlc_row_field_order_locked
    (Write-time guidance #4) so a future T-110b refactor cannot
    silently break this mapping.
    """
    for (symbol, interval), entries in features_by_key.items():
        n = max(feature.warmup_candles for _, feature in entries)
        try:
            async with pool.acquire() as conn:
                rows = await fetch_warmup_window(
                    conn, symbol=symbol, interval=interval, n=n, source=source,
                )
        except Exception as exc:
            logger.error(
                "feature_warmup_load_error",
                symbol=symbol, interval=interval, error=str(exc),
            )
            continue
        for row in rows:
            registry.push(symbol, interval, OhlcCandle(
                symbol=row[0], interval=interval, bucket_start=row[1],
                open=row[2], high=row[3], low=row[4], close=row[5],
                volume=row[6], source=row[7],
            ))
        logger.info(
            "feature_warmup_loaded",
            symbol=symbol, interval=interval, requested=n, loaded=len(rows),
        )
```

### `services/feature_engine/app/main.py` lifespan extension

```python
@asynccontextmanager
async def lifespan(app: FastAPI) -> AsyncIterator[None]:
    """Composition root for async resources (extended from T-109)."""
    features_by_key = build_features()
    capacity_map = {
        key: max(feature.warmup_candles for _, feature in entries)
        for key, entries in features_by_key.items()
    }
    registry = BufferRegistry(capacity_map)

    pool = await create_pool(
        settings.database_url,
        application_name=settings.service_name,
        init=_register_jsonb_codec,
    )
    bus = NatsClient(servers=[settings.nats_url], name=settings.service_name, logger=logger)
    await bus.connect()

    pipeline = FeaturePipeline(
        bus=bus, pool=pool, buffer_registry=registry,
        features_by_key=features_by_key, logger=logger,
    )
    pipeline.acquire_handles()                                  # 1) buffers allocated (refcount 1)
    await warmup_load(
        pool=pool, registry=registry, features_by_key=features_by_key,
        source=SOURCE_BINANCE, logger=logger,                   # 2) buffers warmed in time order
    )
    await pipeline.start_consuming()                            # 3) NATS subscribed; live frames welcome

    app.state.pool = pool
    app.state.bus = bus
    app.state.buffer_registry = registry
    app.state.pipeline = pipeline

    logger.info("service_started", http_port=settings.http_port)
    try:
        yield
    finally:
        await pipeline.stop()                                    # release handles before bus drains
        await bus.close()
        await pool.close()
        logger.info("service_stopped")


async def _register_jsonb_codec(conn: asyncpg.Connection) -> None:
    """JSONB codec init for the pool (T-110b Hand-off option (a)).

    Per Write-time guidance #3: callback contract verified at unit
    level via test_register_jsonb_codec_calls_set_type_codec
    (mock conn; no DB needed).
    """
    await conn.set_type_codec(
        "jsonb", encoder=json.dumps, decoder=json.loads, schema="pg_catalog",
    )
```

## Hand verification — startup ordering trace

Goal: explicit step-by-step verification that the warmup ordering race (Q11) is closed by the `acquire → warmup → subscribe` sequence.

Setup: `features_by_key = {("BTCUSDT", "1m"): [("ind.btcusdt.1m.ema_20", ema20)]}` with `ema20.warmup_candles=20`. Database has 20 prior 1m closes for BTCUSDT.

| step | action                                              | registry state                                           | NATS state                          | invariant check                               |
|------|-----------------------------------------------------|----------------------------------------------------------|-------------------------------------|-----------------------------------------------|
| 1    | `features_by_key = build_features()`                | n/a                                                      | n/a                                 | T-110c contract: substituted feature_name in tuple |
| 2    | `capacity_map = {("BTCUSDT", "1m"): 20}`            | n/a                                                      | n/a                                 | max warmup per key                            |
| 3    | `registry = BufferRegistry(capacity_map)`           | `_counts={}, _buffers={}`                                | n/a                                 | empty (lazy alloc per T-110a)                 |
| 4    | `pool = create_pool(..., init=_register_jsonb_codec)` | n/a                                                    | n/a                                 | JSONB codec on every acquire                  |
| 5    | `bus = NatsClient(...); bus.connect()`              | n/a                                                      | CONNECTED, no subs                  | JS context acquired                           |
| 6    | `pipeline = FeaturePipeline(...)`                   | n/a                                                      | n/a                                 | DI'd; `_handles={}`                           |
| 7    | `pipeline.acquire_handles()` (sync)                 | `_counts={(BTC,1m):1}, _buffers={(BTC,1m):deque(maxlen=20)}` | no subs                          | 0→1 alloc; buffer empty (no live frames yet)  |
| 8    | `warmup_load(..., source=SOURCE_BINANCE)`           | buffer holds 20 rows ASC                                 | no subs                             | live frames could not have arrived (no sub yet); ordering safe |
| 9    | `pipeline.start_consuming()`                        | unchanged                                                | subscribed `market.ohlc.1m.>`        | from this moment live frames push to buffer (deque drops oldest as needed) |
| 10   | live NATS frame X arrives                            | buffer holds 20 rows incl X (oldest dropped)              | sub active                          | X correctly tail-most; warmup data is older   |
| 11   | shutdown begins: `pipeline.stop()`                   | `_counts={}, _buffers={}` (1→0)                          | sub still tracked                   | handles released; refcount 0 deallocs buffer  |
| 12   | `bus.close()`                                       | unchanged                                                | sub drained, NC closed              | NatsClient drains tracked subs                |
| 13   | `pool.close()`                                      | unchanged                                                | unchanged                           | asyncpg pool drained                          |

Critical invariants:
- I1. `acquire_handles` BEFORE `warmup_load` BEFORE `start_consuming` — startup race resolved.
- I2. Warmup pushes only land on already-allocated buffers (Q11 race closed).
- I3. Subscribe activates only after buffers warmed — first live frame hits a populated tail.
- I4. `pipeline.stop()` runs BEFORE `bus.close()` so handle release order matches T-109 lifespan invariant.
- I5. `init=_register_jsonb_codec` callback runs on every `pool.acquire()` so `insert_feature(value_json=...)` writes work without per-call registration.
- I6. `OhlcCandle` mapping passes `interval=interval` from call-site context (T-110b `OhlcRow` tuple lacks the `interval` field; warmup knows interval from the iteration key).
