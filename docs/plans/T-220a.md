@idempotent
async def get_closed_pnl_window(
    self, sub_account: str, since: datetime,
) -> Decimal:
    """OQ-D + ADR-0006 D2-window companion: time-bounded sum.

    Sub_account validation BEFORE limiter.acquire per OQ-10 default A.
    """
    if sub_account != self._sub_account:
        raise ValueError(
            f"sub_account mismatch: got {sub_account!r}, expected {self._sub_account!r}"
        )
    since_ms = int(since.timestamp() * 1000)
    total = Decimal("0")
    cursor: str | None = None
    for _page in range(_MAX_CLOSED_PNL_PAGES):
        params: dict[str, Any] = {
            "category": _CATEGORY,
            "limit": 200,
            "startTime": since_ms,
        }
        if cursor is not None:
            params["cursor"] = cursor
        await self._limiter.acquire(self._sub_account, "positions")
        try:
            result = await self._client.request(
                "GET",
                "/v5/position/closed-pnl",
                params=params,
                retries=3,
            )
        except RateLimitError:
            await self._on_rate_limit_hit("positions")
            raise
        for item in result.get("list", []):
            total += Decimal(item["closedPnl"])
        cursor = result.get("nextPageCursor") or None
        if not cursor:
            return total
    logger.warning(
        "bybit_v5.closed_pnl_window_pagination_capped_at_max_pages",
        extra={"max_pages": _MAX_CLOSED_PNL_PAGES, "sub_account": sub_account, "since": since.isoformat()},
    )
    return total
```

async def sum_paper_trades_realized_pnl_since(
    conn: _DbExecutor,
    *,
    bot_id: str,
    since: datetime,
) -> Decimal:
    """Sum paper_trades.realized_pnl WHERE bot_id=$1 AND status='closed' AND closed_at >= $2.

    Mirror sum_paper_trades_realized_pnl pattern at
    packages/exchange/paper/persistence.py:476 with time-window predicate added.
    Per WG#13: keeps `status = 'closed'` for symmetry with existing helper +
    `closed_at IS NOT NULL` defensive belt-and-suspenders + `closed_at >= since`
    for window filter.
    """
    row = await conn.fetchrow(
        """
        SELECT COALESCE(SUM(realized_pnl), 0) AS total
        FROM paper_trades
        WHERE bot_id = $1
          AND status = 'closed'
          AND closed_at IS NOT NULL
          AND closed_at >= $2
        """,
        bot_id, since,
    )
    return Decimal("0") if row is None else row["total"]
```

async def select_realized_pnl_sum_for_bots_since(
    conn: _DbExecutor,
    *,
    bot_ids: list[str],
    since: datetime,
) -> Decimal:
    """Sum trades.realized_pnl WHERE bot_id IN (...) AND closed_at >= $since.

    T-220b audit uses this for cumulative_db computation per sub-account window.
    Multi-bot list supports cross-bot shared-sub-account composition (per ADR-0004).
    NULL realized_pnl filtered (trades not yet closed by T-219 reconcile_close).
    Empty bot_ids → returns Decimal("0") without DB roundtrip.
    """
    if not bot_ids:
        return Decimal("0")
    row = await conn.fetchrow(
        """
        SELECT COALESCE(SUM(realized_pnl), 0) AS total
        FROM trades
        WHERE bot_id = ANY($1::text[])
          AND closed_at IS NOT NULL
          AND closed_at >= $2
        """,
        bot_ids, since,
    )
    return Decimal("0") if row is None else row["total"]

## Edge cases

- **Empty bot_ids in `select_realized_pnl_sum_for_bots_since`**: short-circuit returns `Decimal("0")` without DB roundtrip; defensive against sub_account with no associated bots in pool (operator misconfiguration).
- **NULL `closed_at` filter**: trades with `realized_pnl IS NULL` (open trades, T-216b1 NULL initial) are EXCLUDED from sum via `closed_at IS NOT NULL` predicate. Symmetric for paper helper.
- **Bybit pagination cap exhaustion**: `_MAX_CLOSED_PNL_PAGES` enforced; logs WARN `bybit_v5.closed_pnl_window_pagination_capped_at_max_pages`. Caller (T-220b) treats partial sum as legitimate input — divergence flag would still fire if cap-truncated sum diverges from DB sum >$0.50.
- **`since` timezone non-UTC**: `int(since.timestamp() * 1000)` is timezone-naive Unix ms; if caller passes naive datetime without tz, undefined behavior. T-220b lifespan threads `now_fn` lambda producing aware UTC datetime per §N1; protocol contract documents UTC requirement.
- **UNIQUE violation on operator-driven concurrent audit run**: per ADR-0007 D7 — let DB raise; T-220b job body catches `UniqueViolationError` + logs WARN + continues to next tick. T-220a's helper does NOT catch — it propagates to caller.

## Write-time guidance (for plan-reviewer carry-forward)

1. **Migration 0009 hypertable + UNIQUE invariant**: composite PK `(audit_run_at, id)` per TimescaleDB convention; UNIQUE `(sub_account, audit_run_at)` per ADR-0007 D7; `id` BIGSERIAL via `GENERATED BY DEFAULT AS IDENTITY` (Postgres 16 idiom; mirror migration 0007 pattern).
2. **Decimal precision invariant**: all NUMERIC(20,4) columns mirror existing `realized_pnl` precision (migration 0005:152). Smoke test in `test_0009_migration.py` writes `Decimal("100.1234")` and asserts SELECT round-trip exact.
3. **`get_closed_pnl_window` UTC contract**: Bybit `startTime` is Unix ms; `int(since.timestamp() * 1000)` converts. Caller MUST pass aware UTC datetime; protocol docstring documents the requirement.
4. **Pagination cap log key**: `bybit_v5.closed_pnl_window_pagination_capped_at_max_pages` (mirror existing `_cumulative_` key naming).
5. **`@non_idempotent` on `insert_trade_pnl_delta`**: per T-213b INSERT precedent; UNIQUE constraint surfaces concurrent-run conflict.
6. **Empty bot_ids short-circuit**: `select_realized_pnl_sum_for_bots_since` returns `Decimal("0")` without DB query when bot_ids list is empty — defensive against sub_account misconfiguration.
7. **`closed_at IS NOT NULL` filter** in both `select_realized_pnl_sum_for_bots_since` (DB side) and `sum_paper_trades_realized_pnl_since` (paper side) — open trades excluded from cumulative sum.
