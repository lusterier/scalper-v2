### `services/feature_engine/app/pipeline.py`

```python
from __future__ import annotations

from datetime import timedelta
from decimal import Decimal
from typing import TYPE_CHECKING, Final

from packages.bus.schemas import (
    FeatureUpdate,
    OhlcCandlePayload,
    subject_for,
)
from packages.bus import MessageEnvelope
from packages.core import CorrelationId
from packages.db.queries.feature_engine import insert_feature
from packages.features.types import OhlcCandle

if TYPE_CHECKING:
    import asyncpg
    from collections.abc import Mapping
    from datetime import datetime
    from structlog.stdlib import BoundLogger

    from packages.bus import NatsClient
    from packages.features.buffers import BufferHandle, BufferRegistry
    from packages.features.protocols import Feature
    from packages.features.types import FeatureValue


_PUBLISHER = "feature-engine"
_KV_BUCKET = "feature_latest"
_SOURCE_BINANCE = "binance"
_SUBJECT_WILDCARD = "market.ohlc.1m.>"

# Bucket-end offset per interval. Pre-emptive forward-compat for F1+
# multi-interval extension (just add 5m/15m/1h/4h/1d keys when
# multi-interval cagg-trigger live updates ship). KeyError on unknown
# interval = fail-loud per §0.4 (T-110c subscribes 1m-only, so unknown
# can only arrive via misconfigured features_registry — surface the bug).
_INTERVAL_DELTA: Final[Mapping[str, timedelta]] = {"1m": timedelta(minutes=1)}


class FeaturePipeline:
    """Per-closed-candle feature dispatch consumer (§9.3, 1m-only).

    Subscribes once to `market.ohlc.1m.>` wildcard. On each closed
    candle: push to the (symbol, "1m") buffer; for each registered
    1m feature, tail buffer → compute → persist → KV update → publish.

    Multi-interval live updates (5m/15m/1h/4h/1d cagg-trigger on 1m
    close) are F1+ deferred (operator D1). Higher-interval features
    in `features_by_key` are populated by T-110d warmup at startup
    but never updated live until F1+ — pipeline holds the buffer
    alive via the BufferHandle so dispatch can still tail() at compute.

    Per-feature error isolation: `except Exception` (NOT bare except)
    so one bad feature doesn't kill the consumer or absorb
    CancelledError. Mirrors T-104b OhlcPipeline `_consume_symbol`
    error semantics.

    Pure-ish consumer class — no I/O resources owned, all DI'd.
    `features_by_key` value tuple is `(feature_name, Feature)` where
    `feature_name` is the **pre-substituted** name from
    `features_registry.build_features()` (single source of truth for
    template substitution; pipeline never re-substitutes).
    """

    def __init__(
        self,
        *,
        bus: NatsClient,
        pool: asyncpg.Pool[asyncpg.Record],
        buffer_registry: BufferRegistry,
        features_by_key: Mapping[tuple[str, str], list[tuple[str, Feature]]],
        logger: BoundLogger,
    ) -> None: ...

    async def start(self) -> None:
        """Acquire BufferHandle per registered (symbol, interval); subscribe.

        Order is load-bearing: handles acquired BEFORE subscribe so the
        first incoming NATS message hits a registered buffer. Empty
        registry → no subscription, no-op.
        """

    async def stop(self) -> None:
        """Release all BufferHandles. Subscription drained by NatsClient.close."""

    async def _on_envelope(self, envelope: MessageEnvelope) -> None:
        """NATS handler — parse payload, route by symbol, dispatch closed candles."""

    async def _dispatch(self, candle: OhlcCandle) -> None:
        """Run (push → compute → persist → KV → publish) per registered 1m feature.

        Iterates `for feature_name, feature in features_by_key.get((symbol, "1m"), [])`.
        Per-feature `except Exception` isolation so one feature's compute
        or persist error doesn't break others.
        """

    async def _persist_kv_publish(
        self,
        feature_name: str,
        feature: Feature,
        candle: OhlcCandle,
        feature_value: FeatureValue,
    ) -> None:
        """DB-then-KV-then-publish; mirror T-104b 'publish-after-persist' contract.

        DB-fail → skip KV+publish (canonical store down → don't notify).
        KV-fail → still publish (NATS+DB stay consistent; KV is best-effort).
        Publish-fail → DB+KV stand (downstream recovers via KV read or
        next compute on the next closed candle).
        """

    @staticmethod
    def _wire_to_domain(payload: OhlcCandlePayload) -> OhlcCandle:
        """Map §8.4 wire payload → §9.3 domain candle (Decimal preserved)."""

    @staticmethod
    def _build_update(
        feature_name: str,
        feature: Feature,
        feature_value: FeatureValue,
        symbol: str,
        candle: OhlcCandle,
    ) -> FeatureUpdate:
        """Map domain FeatureValue → wire FeatureUpdate (Decimal → float seam).

        computed_at = candle.bucket_start + _INTERVAL_DELTA[candle.interval]
        (bucket end; KeyError on unknown interval is intentional fail-loud).

        Single Decimal→float conversion site:
        - value_num: Decimal → float(...)
        - value_bool: pass-through
        - value_json: {k: float(v) if isinstance(v, Decimal) else v for k, v in mapping.items()}

        Non-Decimal sub-values (str, bool, etc.) pass through per L-004
        `Mapping[str, object]` openness (matches T-110b's
        `dict[str, object] | None` Pydantic schema).
        """
```

```python
"""Hardcoded feature registry for T-110c demo (T-111 replaces with YAML loader).

`build_features()` returns Mapping[(symbol, interval), list[tuple[str, Feature]]]
where the leading `str` in each tuple is the **pre-substituted feature_name**.
Substitution happens here at registration time per T-107a docstring
("T-111 substitutes at registration time"); pipeline.py NEVER re-substitutes.

The example feature is EMA-20 on BTCUSDT 1m so the T-110c live path
(1m subscribe → dispatch → persist+KV+publish) is exercised end-to-end.
Higher-interval features can be added here ahead of T-110d's warmup
wiring; their buffers will be populated at startup but not live-updated
until F1+ multi-interval cagg-trigger.

Symbol case in feature_name = lowercase per §1.7 line 244 / §7.2 line 904
/ §8.4 line 1382 example literals (`ind.btcusdt.15m.ema_20`); routing-
key tuple stays canonical Bybit-shape (e.g., `BTCUSDT`) since that's
what `OhlcCandlePayload.symbol` carries.

T-111 will replace this module with a `configs/features/indicators.yaml`
loader behind the same `Mapping[tuple[str, str], list[tuple[str, Feature]]]`
signature so T-110c/T-110d wiring stays untouched at the indicator-
registration swap.
"""

from __future__ import annotations

from typing import TYPE_CHECKING

from packages.features.builtins.ema import EmaFeature

if TYPE_CHECKING:
    from collections.abc import Mapping

    from packages.features.protocols import Feature


__all__ = ["build_features"]


def build_features() -> Mapping[tuple[str, str], list[tuple[str, Feature]]]:
    """Return the demo feature registry with pre-substituted names."""
    symbol = "BTCUSDT"
    interval = "1m"
    feature = EmaFeature(period=20, interval=interval)
    feature_name = feature.name_template.format(
        symbol=symbol.lower(), interval=interval
    )
    return {(symbol, interval): [(feature_name, feature)]}
```

## Decisions committed

1. Subscribe pattern = single wildcard `market.ohlc.1m.>` (one subscription).
2. Symbol routing = `payload.symbol`, NOT subject parsing.
3. Symbol filter = empty `features_by_key.get((symbol, "1m"))` → silent skip + debug log.
4. start() ordering = acquire all BufferHandles → subscribe last.
5. stop() ordering = release handles. Subscription drained by NatsClient.close.
6. Decimal→float conversion site = inline in `_build_update` static method.
7. value_json conversion = iterate sub-keys, Decimal→float, pass-through for bool/str/etc (L-004 openness).
8. **computed_at = `candle.bucket_start + _INTERVAL_DELTA[candle.interval]`** (bucket end via Final mapping; F1+ extension adds 5m/15m/1h/4h/1d keys; KeyError on unknown is intentional fail-loud).
9. correlation_id = `CorrelationId(f"feature:{feature_name}:{symbol}:{computed_at.isoformat()}")`.
10. publisher = `"feature-engine"`.
11. DB write contract = per-feature `pool.acquire()` block.
12. DB-fail / KV-fail / publish-fail handling per "publish-after-persist" contract.
13. KV value encoding = `payload.model_dump_json().encode()`.
14. KV bucket = `"feature_latest"`.
15. KV key format = `f"{feature_name}:{symbol}"` (lowercase in feature_name segment, canonical in suffix per T-110b precedent).
16. Hardcoded example = EMA-20 on BTCUSDT **1m** (live-path exerciser).
17. Source filter = hardcoded `"binance"`.
18. features_registry returns `Mapping[tuple[str, str], list[tuple[str, Feature]]]` API.
19. No subject parsing.
20. In-progress candle = early return on `not is_closed`.
21. FeaturePipeline owns BufferHandle lifecycle.
22. No `_started: bool` guard (minimalist).
23. Q1=a: scalper-v2-features dep added to T-110c pyproject (+~3 LOC delta + uv.lock regen).
24. Test fakes inline in test_pipeline.py.
25. **Template substitution site = `features_registry.build_features()` at registration time**. `Mapping` value tuple `(str, Feature)` carries the substituted feature_name alongside the Feature instance. T-110c pipeline NEVER re-substitutes — single source of truth in the registry layer. T-111 YAML loader inherits the same contract.
26. **Symbol case in substituted feature_name = lowercase** (`symbol.lower()`) per §1.7 line 244 / §7.2 line 904 / §8.4 line 1382 example literals. Routing-key tuple in `features_by_key` and `subject_for(...)` second arg stay canonical Bybit-shape (e.g., `BTCUSDT`).
