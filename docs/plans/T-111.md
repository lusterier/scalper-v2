### `services/feature_engine/app/yaml_loader.py`

```python
"""YAML-driven feature registration for feature-engine (§9.3, §B.2)."""

from __future__ import annotations

from pathlib import Path
from typing import TYPE_CHECKING, Final

import yaml

from packages.features.builtins import (
    AtrFeature, BollingerFeature, EmaFeature, MacdFeature, RsiFeature, VwapFeature,
)

if TYPE_CHECKING:
    from collections.abc import Mapping, Sequence

    from packages.features.protocols import Feature


__all__ = ["load_indicators_yaml"]


_TYPE_TO_FEATURE: Final[Mapping[str, type[Feature]]] = {
    "builtin.ema": EmaFeature,
    "builtin.rsi": RsiFeature,
    "builtin.atr": AtrFeature,
    "builtin.bollinger": BollingerFeature,
    "builtin.macd": MacdFeature,
    "builtin.vwap": VwapFeature,
}


def load_indicators_yaml(
    path: Path,
    symbols: Sequence[str],
) -> Mapping[tuple[str, str], list[tuple[str, Feature]]]:
    """Load configs/features/indicators.yaml and cross-product with symbols.

    Returns Mapping keyed by (symbol, interval) where each value is a list
    of (feature_name, Feature) tuples. feature_name is the pre-substituted
    name_template.format(symbol=symbol.lower(), interval=interval).

    Empty symbols or missing/empty YAML file → empty mapping.

    Raises:
        KeyError: unknown `type:` value in YAML.
        ValueError: substituted name_template or source_version mismatches
            Feature instance attribute (drift check).
    """
    if not path.exists():
        return {}
    raw = yaml.safe_load(path.read_text())
    if not raw or "features" not in raw:
        return {}
    result: dict[tuple[str, str], list[tuple[str, Feature]]] = {}
    for entry in raw["features"]:
        feature_class = _TYPE_TO_FEATURE[entry["type"]]
        interval = entry["interval"]
        params = entry.get("params", {})
        declared_source_version = entry["source_version"]
        declared_template = entry["name_template"]
        for symbol in symbols:
            feature = feature_class(**params, interval=interval)
            if feature.source_version != declared_source_version:
                msg = (
                    f"YAML source_version {declared_source_version!r} "
                    f"mismatches Feature.source_version {feature.source_version!r} "
                    f"for type {entry['type']!r}"
                )
                raise ValueError(msg)
            if feature.name_template != declared_template:
                msg = (
                    f"YAML name_template {declared_template!r} mismatches "
                    f"Feature.name_template {feature.name_template!r} "
                    f"for type {entry['type']!r}"
                )
                raise ValueError(msg)
            feature_name = feature.name_template.format(
                symbol=symbol.lower(), interval=interval
            )
            result.setdefault((symbol, interval), []).append((feature_name, feature))
    return result
```

### `services/feature_engine/app/features_registry.py` (rewrite)

```python
"""YAML-driven feature registry (T-111). Replaces T-110c hardcoded demo."""

from __future__ import annotations

from pathlib import Path
from typing import TYPE_CHECKING

from .yaml_loader import load_indicators_yaml

if TYPE_CHECKING:
    from collections.abc import Mapping, Sequence

    from packages.features.protocols import Feature


__all__ = ["INDICATORS_YAML_PATH", "build_features"]


INDICATORS_YAML_PATH = (
    Path(__file__).resolve().parents[3]
    / "configs"
    / "features"
    / "indicators.yaml"
)


def build_features(
    symbols: Sequence[str],
) -> Mapping[tuple[str, str], list[tuple[str, Feature]]]:
    """Load + cross-product indicators.yaml against `symbols`."""
    return load_indicators_yaml(INDICATORS_YAML_PATH, symbols)
```

## Write-time guidance (must be addressed during implementation)

1. **`build_features` lambda patches v testoch musia akceptovať `symbols` argument**: po zmene signatúry na `(symbols: Sequence[str])` zlyhajú existujúce `lambda: {}` patches s `TypeError: <lambda>() takes 0 positional arguments but 1 was given`. Aktualizovať `services/feature_engine/tests/conftest.py:101` na `lambda symbols: {}` (alebo `lambda *a, **kw: {}` pre defenzívnejšiu variant) AND `services/feature_engine/tests/test_app_factory.py:121` (in `test_jsonb_codec_init_passed_to_create_pool`) na rovnaké. Plán pôvodne označil test_app_factory.py ako "no changes" — to je nepresné, súbor sa musí dotknúť na linke lambda.

2. **PyYAML justification v PR description (§0.6)**: jediná štandardná knižnica pre read-only YAML config loading v Python ekosystéme; alternatívy (ruamel.yaml) sú pre round-trip-preserving — overkill pre náš read-only use-case; `yaml.safe_load` (NIE `yaml.load`) sa nikdy nevolá nad untrusted input (configs sú repo-controlled, nie user-supplied).

3. **Drift check fail messages musia citovať konkrétne dva reťazce ktoré sa rozišli** (YAML hodnotu + Feature instance hodnotu) — diagnostika sa dramaticky degraduje ak ValueError je len `"name_template mismatch"` bez hodnôt. Plán už toto má v Public API skeleton — implementačne držať verbatim. Test #8 (`test_load_yaml_source_version_mismatch_raises_ValueError`) a test #9 (`test_load_yaml_name_template_mismatch_raises_ValueError`) musia asserovať na `match=` regex obsahujúci OBE hodnoty (e.g., `match=r"v2.*v1"` pre source_version).

4. **Empty-file no-op musí byť explicitne testovaný v dvoch scenároch**: súbor neexistuje (test #1) vs súbor je prázdny string (test #11). `yaml.safe_load("")` vracia `None`, NIE `{}`, takže ošetrenie `if not raw` je load-bearing — verify test #11 to pokrýva (`tmp_path / "empty.yaml"; path.write_text(""); assert load_indicators_yaml(path, ["BTC"]) == {}`).

5. **Po implementácii overiť `rg "ind\.\{symbol\}" configs/features/indicators.yaml` má presne 6 hitov**, každý matchujúci §B.2 lines 3045-3084 verbatim. Súbor je shipping data; oprava by si vyžiadala drift hunt cez všetky downstream konzumenty. Zachytiť pri plan-write-time, nie pri brief-reviewer.
